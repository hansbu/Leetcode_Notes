``` python
def search_rotated_sorted_array(self, nums, target):
    l, r = 0, len(nums) - 1
    while l <= r:
        m = l + (r - l)//2
        if target == nums[m]: return m
        if nums[m] <= nums[l]:
            if nums[l] <= target <= nums[m]: r = m - 1
            else: l = m + 1
        else:
            if nums[m] <= target <= nums[r]: l = m + 1
            else: r = m - 1
    return -1

def min_rotated_sorted_array(self, nums):
    l, r = 0, len(nums) - 1
    while l < r:
        m = l + (r - l) // 2
        if nums[m] > nums[r]: l = m + 1
        else: r = m
    return nums[l]
```
